C++中的模板有助于在 C++中实现泛型编程。上一篇教程中的`Swap`函数足够强大，可以支持任何其他需要交换的数据类型。在下面的代码中，`Swap`可以用来交换两个整数。

```
#include <iostream>
using namespace std;

template<typename T>
void Swap(T &a, T&b)
{
    T temp = a;
    a = b;
    b = temp; 
}

int main()
{
    string hello = "world!", world = "Hello, ";
    Swap(world, hello);
    cout << hello << world << endl;

    int a = 5, b = 11;
    Swap(a, b);
    cout << "a:" << a << " b:" << b << endl; 
    return 0;
} 
```
